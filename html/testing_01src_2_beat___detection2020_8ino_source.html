54 void lp(int k) {
55  // Computes the linear prediction weights for the previous values of a given bin k
56  float wk1[LPorder];
57  float wk2[LPorder];
58  float wkm[LPorder];
59 
60  wk1[0] = prevMags[k][0];
61  wk2[LPorder - 2] = prevMags[k][LPorder - 1];
62 
63  for (int i = 1; i < LPorder - 1; ++i) {
64  wk1[i] = prevMags[k][i];
65  wk2[i - 1] = prevMags[k][i];
66  }
67 
68  for (int i = 0; i < LPorder; ++i) {
69  double num = 0.0;
70  double denom = 0.0000000000000001; // to avoid NaNs
71 
72  for (int j = 0; j < (LPorder - i); ++j) {
73  num += wk1[j] * wk2[j];
74  denom += wk1[j] * wk1[j] + wk2[j] * wk2[j];
75  }
76  ks[i] = 2.0 * num / denom;
77 
78  for (int j = 0; j < i; ++j) {
79  ks[j] = wkm[j] - ks[i] * wkm[i - j];
80  }
81  for (int j = 0; j <= i; ++j) {
82  wkm[j] = ks[j];
83  }
84  for (int j = 0; j < (LPorder - i - 1); ++j) {
85  wk1[j] -= wkm[i] * wk2[j];
86  wk2[j] = wk2[j + 1] - wkm[i] * wk1[j + 1];
87  }
88  }
89 }

121  double thr = 0.0; // a threshold to issue a beat, e.g., a frame must be 150% louder than the average

123  int guess = 0;

126  unsigned long currentMillis = millis();
127 
128  // is the FFT available?
129  if (myFFT.available()) {
130 
131  curDel = curDel + 1;
132 
133  // update previous magnitude array
134  for (int i = 1; i < LPorder; i++) {
135  for (int j = 0; j < noBins; j++) {
136  prevMags[j][i - 1] = prevMags[j][i];
137  }
138  }
139 
140  // read current magnitudes
141  for (int i = 0; i < noBins; i++) {
142  n = myFFT.read(i);
143  prevMags[i][LPorder - 1] = n;
144 
145  lp(i); // Compute the weigths for prediction
146 
147  // predict the value of this bin
148  for (int j = 0; j < LPorder; j++) {
149  psd += prevMags[i][j] * ks[j];
150  }
151 
152  // compute the odf with spectral difference and linear prediction
153  sum += abs(n - psd);
154  }
155 
156  // calculate averages:
157  m += ( sum - m ) / numReadings; // running average.
158  // this if implements the copysign function
159  if ((sum - median) >= 0.0 ) {
160  median += m / numReadings;
161  } else {
162  median += -m / numReadings;
163  }
164  //dirty median
165 
166  thr = lambda * median + alpha * m;
167 
168  if ((sum > thr) && (curDel >= minDelay) && (sum >= minLevel)) { // There's a beat
169  avgDelay += ( curDel - avgDelay) / numReadings;
170  curDel = 0; // restart the counter when the delay is at least the average
171 
172  //better delay function, so we do not miss any beats (rob)
173  if (currentMillis - ledStarted >= ONtime_mills) {
174  ledStarted = currentMillis;
175  if (ledState == LOW) {
176  digitalWrite(led, HIGH); // turn the LED on (HIGH is the voltage level)
177  } else {
178  digitalWrite(led, LOW); // turn the LED off by making the voltage LOW
179  }
180  digitalWrite(LED_BUILTIN, ledState);
181  }
182  }
183 
184  if ((curDel >= 2.0 * round(avgDelay)) && (curDel > minDelay) && (sum >= minLevel)) { // There's a beat
185  curDel = 0;
186  guess = 1;
187

189  if (currentMillis - ledStarted >= ONtime_mills) {
