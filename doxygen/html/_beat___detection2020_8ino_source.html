51 void lp(int k) {
52  // Computes the linear prediction weights for the previous values of a given bin k
53  float wk1[LPorder];
54  float wk2[LPorder];
55  float wkm[LPorder];
56 
57  wk1[0] = prevMags[k][0];
58  wk2[LPorder - 2] = prevMags[k][LPorder - 1];
59 
60  for (int i = 1; i < LPorder - 1; ++i) {
61  wk1[i] = prevMags[k][i];
62  wk2[i - 1] = prevMags[k][i];
63  }
64 
65  for (int i = 0; i < LPorder; ++i) {
66  double num = 0.0;
67  double denom = 0.0000000000000001; // to avoid NaNs
68 
69  for (int j = 0; j < (LPorder - i); ++j) {
70  num += wk1[j] * wk2[j];
71  denom += wk1[j] * wk1[j] + wk2[j] * wk2[j];
72  }
73  ks[i] = 2.0 * num / denom;
74 
75  for (int j = 0; j < i; ++j) {
76  ks[j] = wkm[j] - ks[i] * wkm[i - j];
77  }
78  for (int j = 0; j <= i; ++j) {
79  wkm[j] = ks[j];
80  }
81  for (int j = 0; j < (LPorder - i - 1); ++j) {
82  wk1[j] -= wkm[i] * wk2[j];
83  wk2[j] = wk2[j + 1] - wkm[i] * wk1[j + 1];
84  }
85  }
86 }

116 void loop() {
117  double sum = 0.0; // a float to store the sum of all the OFTs bin magnitudes
118  double thr = 0.0; // a threshold to issue a beat, e.g., a frame must be 150% louder than the average
119  double psd = 0.0; //the predicted value
120  int guess = 0;
121 
122  double n; // a float for reading each FFT bin magnitude
123 
124  // is the FFT available?
125  if (myFFT.available()) {
126 
127  curDel = curDel + 1;
128 
129  // update previous magnitude array
130  for (int i = 1; i < LPorder; i++) {
131  for (int j = 0; j < noBins; j++) {
132  prevMags[j][i - 1] = prevMags[j][i];
133  }
134  }
135 
136  // read current magnitudes
137  for (int i = 0; i < noBins; i++) {
138  n = myFFT.read(i);
139  prevMags[i][LPorder - 1] = n;
140 
141  lp(i); // Compute the weigths for prediction
142 
143  // predict the value of this bin
144  for (int j = 0; j < LPorder; j++) {
145  psd += prevMags[i][j] * ks[j];
146  }
147 
148  // compute the odf with spectral difference and linear prediction
149  sum += abs(n - psd);
150  }
151 
152  // calculate averages:
153  m += ( sum - m ) / numReadings; // running average.
154  // this if implements the copysign function
155  if ((sum - median) >= 0.0 ) {
156  median += m / numReadings;
157  } else {
158  median += -m / numReadings;
159  }
160  //dirty median
161 
162  thr = lambda * median + alpha * m;
163 
164  if ((sum > thr) && (curDel >= minDelay) && (sum >= minLevel)) { // There's a beat
165  avgDelay += ( curDel - avgDelay) / numReadings;
166  curDel = 0; // restart the counter when the delay is at least the average
167  digitalWrite(led, HIGH); // turn the LED on (HIGH is the voltage level)
168  delay(ONtime); // wait for a second
169  digitalWrite(led, LOW); // turn the LED off by making the voltage LOW
170  }
171 
172  if ((curDel >= 2.0 * round(avgDelay)) && (curDel > minDelay) && (sum >= minLevel)) { // There's a beat
173  curDel = 0;
174  guess = 1;
175  digitalWrite(led, HIGH); // turn the LED on (HIGH is the voltage level)
176  delay(ONtime); // wait for a second
177  digitalWrite(led, LOW); // turn the LED off by making the voltage LOW
178  }
179  Serial.print(sum, 5);
180  Serial.print("\t");
181  Serial.print(thr, 5);
182  Serial.print("\t");
183  Serial.print(guess == 1 ? -200 : 0);
184  Serial.print("\t");
185  Serial.println(curDel == 0 ? 200 : 0);
186  }
